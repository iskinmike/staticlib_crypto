73  digest_sign_source(Source&& src, const std::string& key_path, const std::string& key_pwd,
74  const std::string& digest_name = "SHA256") :
75  src(std::move(src)) {
76  ctx = std::unique_ptr<EVP_MD_CTX, detail::EVP_MD_CTX_Deleter>(EVP_MD_CTX_create(), detail::EVP_MD_CTX_Deleter());
77  if (nullptr == ctx.get()) {
78  error = -1;
79  return;
80  }
81  auto md = EVP_get_digestbyname(digest_name.c_str());
82  if (nullptr == md) {
83  error = -1;
84  return;
85  }
86  error = EVP_DigestInit_ex(ctx.get(), md, nullptr);
87  if (1 != error) return;
88  // load key
89  auto bio = std::unique_ptr<BIO, detail::BIO_Deleter>(BIO_new(BIO_s_file()), detail::BIO_Deleter());
90  error = BIO_read_filename(bio.get(), key_path.c_str());
91  if (1 != error) return;
92  std::string pwd{key_pwd.data(), key_pwd.length()};
93  void* pwdvoid = static_cast<void*> (std::addressof(pwd.front()));
94  auto key = std::unique_ptr<EVP_PKEY, detail::EVP_PKEY_Deleter>(
95  PEM_read_bio_PrivateKey(bio.get(), nullptr, nullptr, pwdvoid),
96  detail::EVP_PKEY_Deleter());
97  if (nullptr == key.get()) {
98  error = -1;
99  return;
100  }
101  error = EVP_DigestSignInit(ctx.get(), nullptr, md, nullptr, key.get());
102  }

170  const std::string& get_signature() {
171  if (1 == error && signature.empty()) {
172  size_t req = 0;
173  error = EVP_DigestSignFinal(ctx.get(), nullptr, std::addressof(req));
174  if (1 == error && req > 0) {
175  std::string sig;
176  sig.resize(req);
177  size_t slen = req;
178  error = EVP_DigestSignFinal(ctx.get(), reinterpret_cast<unsigned char*> (std::addressof(sig.front())),
179  std::addressof(slen));
180  if (1 == error && req == slen) {
181  signature = to_hex(sig);
182  }
183  }
184  }
